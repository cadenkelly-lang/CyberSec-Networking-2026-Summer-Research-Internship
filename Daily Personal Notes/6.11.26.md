cat > /etc/nftables.conf <<'EOF'
#!/usr/sbin/nft -f

flush ruleset

define WAN = ens18

define INTERNAL_IFACES = {
    ens19, ens20, ens21,
    ens22, ens23, ens1,
    ens2, ens3, ens4,
    ens5, ens6, ens7
}

table inet filter {
    chain input {
        type filter hook input priority 0;
        policy drop;

        iif lo accept
        ct state established,related accept

        # Allow ping to this router
        ip protocol icmp accept

        # Allow DHCP requests from VLANs 20+
        iifname $INTERNAL_IFACES udp dport { 67, 68 } accept

        # Allow SSH management from VLAN 10 only
        iifname $WAN tcp dport 22 accept
    }

    chain forward {
        type filter hook forward priority 0;
        policy drop;

        ct state established,related accept

        # Allow VLANs 20+ out to VLAN 10 / upstream gateway
        iifname $INTERNAL_IFACES oifname $WAN accept

        # Allow VLAN 10 to reach internal VLANs for admin/testing
        iifname $WAN oifname $INTERNAL_IFACES accept
    }

    chain output {
        type filter hook output priority 0;
        policy accept;
    }
}

table ip nat {
    chain postrouting {
        type nat hook postrouting priority 100;

        # NAT VLANs 20+ behind 192.168.10.220 when going out via VLAN 10
        iifname $INTERNAL_IFACES oifname $WAN masquerade
    }
}
EOF

#!/usr/sbin/nft -f

flush ruleset

define WAN = ens18

define INTERNAL_IFACES = {
    ens19, ens20, ens21,
    ens22, ens23, ens1,
    ens2, ens3, ens4,
    ens5, ens6, ens7
}

table inet filter {
    chain input {
        type filter hook input priority 0;
        policy drop;

        iif lo accept
        ct state established,related accept

        # Allow ping to this router
        ip protocol icmp accept

        # Allow DHCP requests from VLANs 20+
        iifname $INTERNAL_IFACES udp dport { 67, 68 } accept

        # Allow SSH management from VLAN 10 only
        iifname $WAN tcp dport 22 accept
    }

    chain forward {
        type filter hook forward priority 0;
        policy drop;

        ct state established,related accept

        # Allow VLANs 20+ out to VLAN 10 / upstream gateway
        iifname $INTERNAL_IFACES oifname $WAN accept

        # Allow VLAN 10 to reach internal VLANs for admin/testing
        iifname $WAN oifname $INTERNAL_IFACES accept
    }

    chain output {
        type filter hook output priority 0;
        policy accept;
    }
}

table ip nat {
    chain postrouting {
        type nat hook postrouting priority 100;

        # NAT VLANs 20+ behind 192.168.10.220 when going out via VLAN 10
        iifname $INTERNAL_IFACES oifname $WAN masquerade
    }
}
